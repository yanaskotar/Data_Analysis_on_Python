years = range(1880, 2018)

pieces = []
columns = ['name', 'sex', 'births']
for year in years:
    path = 'data/babynames/yob{}.txt'.format(year)
    frame = pd.read_csv(path, names=columns)
    frame['year'] = year
    pieces.append(frame)

names = pd.concat(pieces, ignore_index=True)
